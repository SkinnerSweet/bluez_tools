62 int8_t l2cap_server_create(l2cap_server_t *server, bt_address_t *adapter, uint16_t port,
63  uint8_t max_clients, uint16_t buffer_length,
64  void (*treat_buffer_func)(l2cap_server_t *server, uint8_t num_client),
65  void (*send_response_func)(l2cap_server_t *server, uint8_t num_client, uint8_t res_type)) {
66 
67  if (!server) {
68  print_trace(TRACE_ERROR, "l2cap_server_create : invalid server reference.\n");
69  return -1;
70  }
71 
72  if (!adapter) {
73  print_trace(TRACE_ERROR, "l2cap_server_create : invalid BT adapter reference.\n");
74  return -1;
75  }
76 
77  if (!buffer_length) {
78  print_trace(TRACE_WARNING, "l2cap_server_create : null buffer, creation aborted.\n");
79  return -1;
80  }
81 
82  server->launched = 0;
83  server->socket = open_l2cap_socket(adapter, port, 1);
84  if (server->socket.sock < 0) {
85  print_trace(TRACE_ERROR, "Unable to create listening socket on server.\n");
86  return -1;
87  }
88  server->buffer_length = buffer_length;
89  server->max_clients = max_clients;
90  server->clients = calloc(max_clients, sizeof(l2cap_client_proxy_t));
91  for (uint8_t i = 0; i < max_clients; i++) {
92  server->clients[i].buffer = calloc(buffer_length, sizeof(char));
93  }
94 
95  if (!treat_buffer_func) {
96  server->treat_buffer = &(treat_buffer_default_func);
97  } else {
98  server->treat_buffer = treat_buffer_func;
99  }
100  if (!send_response_func) {
101  server->send_response = &(send_response_default_func);
102  } else {
103  server->send_response = send_response_func;
104  }
105 
106  return 0;
107 }

111 static void *server_thread_routine(void *data) {
112  struct routine_data_t *routine_data = (struct routine_data_t *)data;
113 
114  uint8_t i = routine_data->num_client;
115  l2cap_server_t *server = routine_data->server;
116  l2cap_client_proxy_t my_client = server->clients[i];
117  int16_t timeout = routine_data->timeout;
118  uint16_t max_req = routine_data->max_req;
119  char *buffer = server->clients[i].buffer;
120 
121  char thread_launched = 1;
122  uint16_t num_req = 0;
123 
124  while (thread_launched && ((num_req < max_req) || max_req == -1)) {
125  memset(buffer, 0, server->buffer_length);
126  int8_t bytes_read = 0;
127  int8_t n = 0;
128  struct pollfd p;
129  p.fd = my_client.conn_id;
130  p.events = POLLIN;
131 
132  while ((n = poll(&p, 1, timeout)) < 0) {
133  if (errno == EAGAIN || errno == EINTR) {
134  continue;
135  }
136  perror("server_routine : error while polling socket");
137  goto end;
138  }
139 
140  if (!n) {
141  errno = ETIMEDOUT;
142  perror("server_routine : error while polling socket");
143  goto end;
144  }
145 
146  while ((bytes_read = read(my_client.conn_id,
147  buffer,
148  server->buffer_length)) < 0) {
149  if (errno == EAGAIN || errno == EINTR)
150  continue;
151  perror("server_routine : error while reading socket.\n");
152  goto end;
153  }
154 
155  if (bytes_read == 0) { // 0 Bytes read means that the connection has been lost.
156  print_trace(TRACE_WARNING, "l2cap_server : nothing to read on the socket.\n");
157  goto end;
158  }
159 
160  num_req++;
161 
162  if (strcmp(buffer, L2CAP_SERVER_UNIVERSAL_STOP) == 0) {
163  char closeACK[10] = "STOP_ACK";
164  if (write(my_client.conn_id, closeACK, 10) < 10) {
165  print_trace(TRACE_WARNING, "l2cap_server : unable to send STOP_ACK.\n");
166  }
167  thread_launched = 0;
168  }
169  server->treat_buffer(server, i);
170  }
171 
172  end:
173  print_trace(TRACE_INFO, "l2cap_server : connection %i ended.\n", my_client.conn_id);
174  close(my_client.conn_id);
175 
176  return NULL;
177 }

181 int8_t l2cap_server_launch(l2cap_server_t *server, int16_t timeout, uint16_t max_req) {
182  if (server == NULL) {
183  print_trace(TRACE_ERROR, "l2cap_server_launch : invalid server reference.\n");
184  return -1;
185  }
186 
187  if (server->max_clients == 0) {
188  print_trace(TRACE_ERROR, "l2cap_server_launch : no socket available to run the server.\n");
189  return -1;
190  }
191 
192  if (server->socket.sock < 0) {
193  print_trace(TRACE_ERROR, "server_launch_socket_routine : invalid socket.\n");
194  return -1;
195  }
196 
197  pthread_t server_threads[server->max_clients];
198  struct routine_data_t routine_data[server->max_clients];
199  for (uint8_t i = 0; i < server->max_clients; i++) {
200  routine_data[i].server = server;
201  routine_data[i].timeout = timeout;
202  routine_data[i].max_req = max_req;
203  routine_data[i].num_client = i;
204  }
205 
206  int8_t sock = server->socket.sock;
207  socklen_t sockaddr_len = sizeof(l2cap_sockaddr_t);
208  int8_t client_i = 0;
209  int client_sock = 0;
210 
211  listen(sock, server->max_clients);
212  print_trace(TRACE_INFO, "Waiting for connection on socket %i...\n", sock);
213 
214  // Now that the socket is listening, we check connection demands
215  struct pollfd p;
216  int8_t n = 0;
217  p.fd = sock;
218  p.events = POLLIN;
219 
220  while ((n = poll(&p, 1, timeout)) < 0) {
221  if (errno == EAGAIN || errno == EWOULDBLOCK)
222  continue;
223  perror("server_routine_accept : error while polling socket");
224  goto fail;
225  }
226 
227  if (!n) {
228  errno = ETIMEDOUT;
229  perror("server_routine_accept : error while polling socket");
230  goto fail;
231  }
232 
233  // We create a new thread for each client accepted
234  while ((client_sock = accept(sock,
235  (struct sockaddr *)&(server->clients[client_i].rem_addr),
236  &sockaddr_len))) {
237 
238  print_trace(TRACE_INFO, "Connection established on socket %i, connection id : %i...\n", sock, client_sock);
239  server->clients[client_i].conn_id = client_sock;
240 
241  if(pthread_create(&(server_threads[client_i]), NULL, &(server_thread_routine), (void *)&routine_data[client_i]) < 0) {
242  perror("could not create thread");
243  goto fail;
244  }
245  client_i++;
246  server->launched = 1;
247  }
248 
249  if (client_sock < 0) {
250  perror("accept failed");
251  goto fail;
252  }
253 
254  for (uint8_t i = 0; i < client_i; i++) {
255  pthread_join(server_threads[i], NULL);
256  }
257 
258  return 0;
259 fail:
260  return -1;
261 }
