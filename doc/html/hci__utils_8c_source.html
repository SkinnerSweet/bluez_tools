69 void hci_compute_filter(struct hci_filter *flt, ...) {
70  va_list pa;
71  uint32_t event;
72 
73  if (flt == NULL) {
74  print_trace(TRACE_ERROR, "hci_compute_filter : invalid filter.\n");
75  return;
76  }
77 
78  hci_filter_clear(flt);
79  hci_filter_set_ptype(HCI_EVENT_PKT, flt);
80  // HCI_EVENT_PKT = HCI Event Packet.
81 
82  va_start(pa, flt);
83  event = va_arg(pa, uint32_t);
84  while (event != 0) {
85  hci_filter_set_event(event, flt);
86  event = va_arg(pa, uint32_t);
87  }
88  va_end(pa);
89 }

93 void hci_display_LE_supported_states(uint64_t states) { // p1096
94  fprintf(stdout, "Device supported states : 0x%016llX\n", (long long unsigned int)states);
95  uint8_t offset = 0;
96  while (offset < 29) {
97  if (states & states_map[offset].mask) {
98  fprintf(stdout, " -> %s supported. \n", states_map[offset].description);
99  } else {
100  fprintf(stdout, " -> %s UNsupported. \n", states_map[offset].description);
101  }
102 
103  offset++;
104  }
105 }
