43 static void send_request_default_func(l2cap_client_t client, uint8_t req_type) {
44  if (client.l2cap_socket.sock < 0) {
45  print_trace(TRACE_ERROR, "client_send_request : invalid socket descriptor.\n");
46  return;
47  }
48  char request[] = "Request echo.";
49  if (write(client.l2cap_socket.sock, request, 14) < 14) {
50  print_trace(TRACE_ERROR, "Unable to write request.\n");
51  }
52  return;
53 }

57 int8_t l2cap_client_create(l2cap_client_t *client, bt_address_t *server_add, uint16_t port,
58  uint16_t buffer_length,
59  void (*treat_buffer_func)(l2cap_client_t client),
60  void (*send_request_func)(l2cap_client_t client, uint8_t req_type)) {
61 
62  if (!client) {
63  print_trace(TRACE_ERROR, "l2cap_client_create : invalid client reference.\n");
64  return -1;
65  }
66 
67  if (!buffer_length) {
68  print_trace(TRACE_WARNING, "l2cap_client_create : null buffer, creation aborted.\n");
69  return -1;
70  }
71 
72  client->connected = 0;
73  client->l2cap_socket = open_l2cap_socket(server_add, port, 0);
74  if (client->l2cap_socket.sock < 0) {
75  print_trace(TRACE_ERROR, "l2cap_client_create : cannot open socket.\n");
76  return -1;
77  }
78  client->buffer = calloc(buffer_length, sizeof(char));
79  client->buffer_length = buffer_length;
80 
81  if (!treat_buffer_func) {
82  client->treat_buffer = &(treat_buffer_default_func);
83  } else {
84  client->treat_buffer = treat_buffer_func;
85  }
86  if (!send_request_func) {
87  client->send_request = &(send_request_default_func);
88  } else {
89  client->send_request = send_request_func;
90  }
91 
92  return 0;
93 }

97 int8_t l2cap_client_connect(l2cap_client_t *client) {
98  int8_t status = -1;
99  status = connect(client->l2cap_socket.sock,
100  (struct sockaddr *)&(client->l2cap_socket.sockaddr),
101  sizeof(client->l2cap_socket.sockaddr));
102  client->connected = (status == 0);
103 
104  return status;
105 }

109 int8_t l2cap_client_send(l2cap_client_t *client, int16_t timeout, uint8_t req_type) {
110  if (!client) {
111  print_trace(TRACE_ERROR, "l2cap_client_send : invalid client.\n");
112  return -1;
113  }
114 
115  if (client->connected) {
116  memset(client->buffer, 0, strlen(client->buffer));
117  client->send_request(*client, req_type);
118  int8_t n = 0;
119  int8_t bytes_read;
120  struct pollfd p;
121  p.fd = client->l2cap_socket.sock;
122  p.events = POLLIN;
123 
124  while ((n = poll(&p, 1, timeout)) < 0) {
125  if (errno == EAGAIN || errno == EINTR) {
126  continue;
127  }
128  if (errno == ENOTCONN) {
129  client->connected = 0;
130  l2cap_client_close(client);
131  }
132  perror("l2cap_client_send : error while polling socket");
133  return -1;
134  }
135 
136  if (!n) {
137  errno = ETIMEDOUT;
138  perror("l2cap_client_send : error while polling socket");
139  return -1;
140  }
141 
142  while((bytes_read = read(client->l2cap_socket.sock,
143  client->buffer,
144  client->buffer_length)) < 0) {
145  if (errno == EAGAIN || errno == EINTR)
146  continue;
147  if (errno == ENOTCONN) {
148  client->connected = 0;
149  l2cap_client_close(client);
150  }
151  perror("l2cap_client_send : error while reading socket.\n");
152  return -1;
153  }
154 
155  if (bytes_read == 0) { // 0 Bytes read means that the connection has been lost.
156  print_trace(TRACE_WARNING, "l2cap_client: connection reset by peer.\n");
157  client->connected = 0;
158  return -1;
159  }
160 
161  client->treat_buffer(*client);
162 
163  } else {
164  print_trace(TRACE_ERROR, "l2cap_client_send : invalid connexion.\n");
165  return -1;
166  }
167 
168  return 0;
169 }
