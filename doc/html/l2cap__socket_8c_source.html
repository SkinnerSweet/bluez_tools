33 // If controler == NULL, the first present available BT adaptator is taken.
34 l2cap_socket_t open_l2cap_socket(bt_address_t *adapter, uint16_t port, char to_bind) {
35  l2cap_socket_t result;
36 
37  memset(&result, 0, sizeof(result));
38 
39  (result.sockaddr).l2_family = AF_BLUETOOTH;
40  (result.sockaddr).l2_psm = htobs(port);
41 
42  if (!adapter) {
43  adapter = BDADDR_ANY;
44  }
45 
46  (result.sockaddr).l2_bdaddr = *adapter;
47  //result.l2_cid ???
48 
49  result.sock = socket(AF_BLUETOOTH, SOCK_SEQPACKET, BTPROTO_L2CAP);
50  if (result.sock < 0) {
51  perror("opening socket");
52  result.sock = -1;
53  return result;
54  }
55 
56  if (to_bind) {
57  if (bind(result.sock, (const struct sockaddr *)&(result.sockaddr), sizeof(result.sockaddr)) < 0) {
58  perror("binding socket");
59  close(result.sock);
60  result.sock = -1;
61  return result;
62  }
63  }
64 
65  list_push(&l2cap_socket_list, &result, sizeof(l2cap_socket_t));
66 
67  return result;
68 }

72 void close_l2cap_socket(l2cap_socket_t *l2cap_socket) {
73  if (l2cap_socket->sock < 0) {
74  print_trace(TRACE_WARNING, "close_l2cap_socket : already closed socket.\n");
75  return;
76  }
77  close(l2cap_socket->sock);
78  l2cap_socket_t *listed_socket = list_search(&l2cap_socket_list,
79  (const void *)l2cap_socket,
80  sizeof(l2cap_socket_t));
81  l2cap_socket->sock = -1;
82  if (listed_socket == NULL) {
83  print_trace(TRACE_WARNING, "close_l2cap_scoket : this socket wasn't referenced yet.\n");
84  return;
85  }
86  free(listed_socket);
87  return;
88 }

98 void close_all_l2cap_sockets(void) {
99  if (l2cap_socket_list == NULL) {
100  print_trace(TRACE_ERROR, "close_all_l2cap_sockets : no socket to close.\n");
101  return;
102  }
103 
104  l2cap_socket_t *l2cap_socket = NULL;
105 
106  while (l2cap_socket_list != NULL) {
107  l2cap_socket = (l2cap_socket_t *)list_pop(&l2cap_socket_list);
108  if (l2cap_socket->sock >= 0) {
109  close(l2cap_socket->sock);
110  }
111  free(l2cap_socket);
112  }
113 }
